 * Purpose   : Perfect Squares

 */
 
:- module(squares, [squares/1]).

:- use_module(library(lists)).
:- use_module(library(clpfd)).

squares(ID) :-
	findall(1, (constraints(ID, Xs, Ys, Sizes),search(cp(0.5),Xs,Ys,Sizes)), L),
	length(L, N),
	data(ID, Nsq, WH, DSizes),
	format('number of solutions = ~d\n', [N]),
	format('placement space = ~dx~d\n', [WH,WH]),
	format('~d squares of size = ~w\n', [Nsq,DSizes]).

search(cp(W), Xs, Ys, Sizes) :-
	by_intervals(Xs, Sizes, W),
	labeling([bisect], Xs),
	by_intervals(Ys, Sizes, W),
	labeling([bisect], Ys).

constraints(ID, Xs, Ys, Sizes) :-
	data(ID, _, Size, Data),
	sort(Data, Rev),
	reverse(Rev, Sizes),
	(   foreach(X,Xs),
	    foreach(Y,Ys),
	    foreach(S,Sizes),
	    foreach(S-[X,Y],L1),
	    foreach(object(I,I,[X,Y]),Objs),
	    foreach(sbox(I,[0,0],[S,S]),Shapes),
	    count(I,1,_),
	    param(Size)
	do  Sd is Size-S+1,
	    X in 1..Sd,
	    Y in 1..Sd
	),
	keysort(L1, L2),
	keyclumped(L2, L3),
	(   foreach(_-L,L3)
	do  (L=[_] -> true ; lex_chain(L, [op(#<)]))
	),
	geost(Objs, Shapes, [cumulative(true),
			     disjunctive(true),
			     longest_hole(true,1000),
			     parconflict(true),
			     visavis_init(true), 
			     dynamic_programming(true)]).

by_intervals([], _, _).
by_intervals([X|Xs], [N|Ns], W) :-
	by_interval(X, N, W),
	by_intervals(Xs, Ns, W).

by_interval(X, _,   _) :-
	integer(X), !.
by_interval(X, Len, W) :-
	fd_min(X, Min),
	fd_max(X, Max),
	Mid is (Min+Max)>>1,
	(   Max-Min < Len*W -> true
	;   X in Min..Mid,
	    by_interval(X, Len, W)
	;   Mid1 is Mid+1,
	    X in Mid1..Max,
	    by_interval(X, Len, W)
	).
